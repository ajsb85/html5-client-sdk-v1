JS/HTML Tutorial

Here a brief instructions for using the LogoSDK on HTML.
Please remember to show the console navigator.

In the HTML/JSP file:
<html> <head> <meta http-equiv="Content-Type" content="text/html; charset=UTF-8"> <title>tester logo API</title> <script type="text/javascript" src="js/LogoSDK.js"></script> <script type="text/javascript" src="js/code.js"></script> </head> <body> <h1>Other actions</h1> <button id='clearButton' style="float: left;"> Clear logo saved </button> <div id='ClearCache'></div> <br/> <h1>Logo tests:</h1> <button id="getLogoButton"> Test Logo </button> <button id="getSmallLogoButton"> Test small Logo </button> </body> </html>
In the Javascript code.js file:
var serviceUri="https://integration-axiatagroup.apiexchange.org/v1/logo"; var mcc = "214"; var mnc = "18"; var sourceIP = null; var logosize = null; var bg_color = null; var aspect_radio = null; window.onload = function(e){ document.getElementById("clearButton").onclick = function(){ clearCacheLogo(); document.getElementById("ClearCache").innerHTML="Cleared!! "; }; document.getElementById("getLogoButton").onclick = function(){ getLogo(serviceUri, mcc, mnc, sourceIP, logosize, bg_color, aspect_ratio, callbackLogoFunction); }; mcc = null; mnc = null; logosize = "small"; document.getElementById("getSmallLogoButton").onclick = function(){ getLogo(serviceUri, mcc, mnc, sourceIP, logosize, bg_color, aspect_ratio, callbackLogoFunction); }; }; function callbackLogoFunction(result) { if(result.error===undefined){ console.log('logo Result: '+JSON.stringify(result)); for ( var int = 0; int < result.getLogos().length; int++) { console.log(int+" >> "+JSON.stringify(result.getLogos()[int])); console.log('img url '+result.getLogos()[int].getUrl()); } }else{ console.log("ERROR: "+result.error+": ", result.error_description); } }
